############# LISTAS DE CONTROL DE ACCESO ################

# VERIFICAR SI EL ARCHIVO O CARPETA TIENE UNA ACL
ls -la
getfacl -c acl_ocantoran

# ASIGNAR PERMISOS DE LECTURA ESCRITURA A TRAVES DE UNA ACL DE ACCESO PARA EL USUARIO ocantoran Y UTILIZANDO LA NOTACIÃ“N OCTAL
setfacl -m u:ocantoran:6 acl_ocantoran

# GENERAR UNA NUEVA LISTA DE ACCESO PARA EL USUARIO victor Y ASIGNAR PERMISOS DE rwx MEDIANTE NOTACION SIMBOLICA
(ESTO HACE QUE LA MASCARA SE MODIFIQUE PARA ADAPTARSE AL MAXIMO NIVEL DE PERMISOS POSIBLE QUE AHORA ES rwx)
setfacl -m u:victor:rwx acl_ocantoran

# ELIMINAR LA LISTA DE ACCESO SOBRE EL USUARIO VICTOR Y VERIFIQUE QUE LA MASK SE HAYA ACTUALIZADO
setfacl -x u:victor acl_ocantoran

# ELIMINAR TODAS LAS ENTRADAS DE ACL EXTENDIDAS O ACL DE ACCESO
setfacl -b acl_ocantoran

########################## ACL POR DEFECTO #####################################
# ESTAS SOLO SE PUEDEN HACER EN DIRECTORIOS Y SON RECURSIVAS, PERO AL MOMENTO DE ELIMINARLAS NO LO HACE DE MANERA RECURSIVA

# GENERE IMA ;OSTA DE ACCESO CON PERMISOS rwx PARA LOS USUARIOS angel Y victor (LA OPCION -d ES DE DEFAULT)
setfacl -dm u:angel:7,u:victor:rwx proyectos
