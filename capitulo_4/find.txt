# ENCONTRAR TODOS LOS ARCHIVOS CUYO USUARIO PROPIETARIO SEA ROOT Y SU GRUPO PROPIETARIO SEA CUALQUIERA EXCEPTO ROOT
find / -user root -not -group root
find / -user root ! -group root 

# ENCONTRAR TODOS LOS ELEMENTOS DE TIPO DIRECTORIO CUYO NOMBRE SEA src UBICADOS EN /usr Y CON UN MAXIMO DE DOS NIVELES DE PROFUNDIDAD
find /usr -maxdepth 2 -type d -name src
# ejemplo de niveles de profundidad que son 2  /usr/local/src 

# MINIMO 3 NIVELES DE PROFUNDIDAD
find /usr -mindepth 3 -type d -name src
# ejemplo de niveles de profundidad que son 3 /usr/local/otra/src

# ENCONTRAR EN EL DIRECTORIO /etc TODOS LOS ARCHIVOS QUE HAYAN SIDO MODIFICADOS HACE MAS DE 2000 DIAS
find /etc/ -mtime +2000

# ENCONTRAR LOS ARCHIVOS QUE HAN SIDO MODIFICADOS DURANTE LOS ULTIMOS 100 MINUTOS O MENOS EN EL DIRECTORIO /var/log
find /var/log -mmin -100

# ENCONTRAR ARCHIVOS DE DISPOSITIVO DE BLOQUE EN EL DIRECTORO /dev QUE TENGAN ESTABLECIDOS LOS PERMISOS 660
find /dev -type b -perm 660

# ENCONTRAR ARCHIVO DE TIPO CARACTER (c) EN /dev CON ALMENOS PERMISOS DE ESCRITURA ES DECIR PERMISOS 222 COMO MINIMO AQUI SE IGNORA LOS PERMISOS rx
find /dev -type c -perm -2222

# ENCONTRAR TODOS LOS ARCHIVOS EN /etc/systemd QUE TENGAN ALMENOS PERMISOS DE EJECUCION POR EL PROPIETARIO Y POR LOS MIEMBROS DEL GRUPO
find /etc/systemd -perm /110

# ENCUENTRA TODOS LOS ARCHIVOS DE TIPO LINK SIMBOLICO (l) EN EL DIRECTORIO /usr QUE TENGAN POR LO MENOS PERMISOS DE LECTURA Y ESCRITURA PARA EL PROPIETARIO Y EL GRUPO PROPIETARIO 
find /usr -type l -perm -ug=rw

# BUSCAR EN TODO EL SISTRMA TODOS LOS ARCHIVOS QUE SE LLAMEN CORE Y DESPUES QUE EJECUTE UN ls -ld
find / -name core -type d -exec ls -ld {} \;

# BUSCAR TODOS LOS ARCHIVOS QUE SE LLAMEN EJEMPLO EN practica8 Y COPIARLOS A /tmp 
find /root/Documentos/practica8 -iname ejemplo* -ok cp {} /tmp \;
