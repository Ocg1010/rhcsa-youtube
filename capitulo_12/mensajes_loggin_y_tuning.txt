SYSTEMD JORUNAL
# EJECUTE EL COMANDO journalctl PARA MIRAR TODOS LOS LOGS GENERADOS A PARTIR DEL ULTIMO REINICIO DEL SISTEMA 
journalctl

# GENERAR UN REPORTE DE LOGS DONDE LA SALIDA SEA DETALLADA 
journalctl -o verbose

# MIRAR TODOS LOS EVENTOS GENERADOS DESDE EL ÚLTIMO REINICIO DEL SISTEMA 
journalctl -b

# MIRAR TODOS LOS EVENTOS GENERADOS DESDE LOS ULTIMOS DOS REINICIOS DEL SISTEMA 
journalctl -kb2

# LIMITAR LA CANTIDAD DE REGISTROS MOSTRADOS EN LA SALIDA A SOLO 3 
jorunalctl -n3

# MOSTRAR TODAS LAS ALERTAS GENERADAS POR UN SERVICIO PARTICULAR EJEMPLO crond 
journalctl /usr/sbin/crond
witch crond (para encontrar una ruta de un binario)

# RECUPERAR TODOS LOS MENSAJES DE LOG GENERADOS POR UN EVENTO QUE ESTA ASOCIADO CON UN PID QUE PERTENEZCA AL SERVICIO chroyd
journalctl _PID=$(pgrep chronyd)

# MOSTRAR TODOS LOS MENSAJES DE LOG GENERADOS POR UN SYSTEM UNIT EN PARTICULAR EJEMPLO sshd.service 
journalctl _SYSTSEM_UNIT=sshd.service

# MOSTRAR TODOS LOS MENSAJES DE ERROR GENERADOS EN UN DETERMINADO RANGO DE FECHAS POR EJEMPLO  
journalctl --since 2022-06-01 --until 2022-06-20 -p err

# OBTENER TODOS LOS MENSAJES DE ADVERTENCIA (warnings) GENERADOS EN EL DÍA DE HOY Y MOSTRARLOS EN ORDEN CRONOLOGICO A LA INVERSA (EL ULTIMO SE MUESTRA PRIMERO)
journalctl --since today -p warning -r
yesterday, today, tomorrow

# MIRAR LOS LOGS EN TIEMPO REAL 
journalctl -f


PRESERVAR LA INFORMACION DEL JOURNALCTL
# CREAR EL DIRECTORIO /var/log/journal PARA QUE LOS DIRECTORIOS DE journalctl SE GUARDEN EN FORMA PERSISTENTE 
mkdir /var/log/journal



SYSTEM TUNING
# LISTAR LOS PERFILES PREDEFINIDOS
ls -l /usr/lib/tuned

# INSTALAR LOS PAQUETES EN LA APLICACIÓN tuned 
dnf install tuned

# ARRANQUE EL SERVICIO tuned Y ESTABLEZCA EL AUTOARRANQUE CON CADA REINICIO  
sudo systemctl --now enable tuned

# VERIFIQUE EL ESTATUS  
systemctl status tuned

# DESPLIEGUE LA LISTA DE PERFILES DE tuning DISPONIBLE PARA VER LOS 9 PERFILES DISPONIBLES ASÍ COMO EL PERFIL ACTUAL 
tuned-adm list

# CAMBIE AL PERFIL POWERSAVE Y CONFIRME QUE ESTA ÁCTIVO  
tunes-adm profile powersave
tunes-adm active

# DETERMINE CÚAL ES EL PERFIL RECOMENDADO PARA SU SERVIDOR Y PROCEDA A REALIZAR EL CAMBIO 
tuned-adm recommend
tuned-adm profile virtual-guest
tuned-adm active

# APAGUE EL TUNING  
tuned-adm off
tuned-adm active

# VUELVA A ACTIVAR EL tunning Y CONFIRME
tuned-adm profile virtual-guest
tunes-adm active
