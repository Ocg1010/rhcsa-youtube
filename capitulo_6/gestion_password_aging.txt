# DESHABILITAR UNA CUENTA Y BLOQUEAR TIENEN DIFERENCIA
# DESHABILITAR NORMALMENTE SE HACE CUANDO UN USUARIO CAUSA BAJA EN LA EMPRESA Y BLOQUEAR SE HACE CUANDO UN USUARIO EXCEDE EL INTENTO DEFINIDO DE INICIO DE SESION POR CONTRASEÃ‘A INCORRECTA 

-d --lastdat: NUMERO DE DIAS TRANSCURRIDOS DESDE QUE EL PASSWORD CAMBIO POR ULTIMA VEZ, UNIX TIME; CAMPO 3 DEL ARCHIVO /etc/shadow 
-d 0 : CAMBIO OBLIGATORIO DE PASSWD AL SIGUIENTE INICIO DE SESION
-E: DEFINE FECHA DE EXPIRACION fecha AAAA-MM-DD o numero de dias que es passwd estara
-E -1: DESACTIVAMOS ESA CARACTERISTICA (EL PASSWD NO TENDRIA FECHA DE EXPIRACION)
-I: DEFINE NUMERO DE DIAS QUE LA CUENTA ESTARA DESACTIVADA DESPUES DE QUE EL PASSWORD CADUQUE 
-M: MAXIMO DE DIAS EN QUE EL PASSWD ES VALIDO, ANTES DE QUE SE DEBA CAMBIAR
-m MINIMO DE DIAS TRANSCURRIDOS PARA QUE EL USUARIO PUEDA CAMBIAR EL PASSWORD
-W: CANTDAD DE DIAS EN QUE EL USUARIO RECIBIRA AVISOS DE CAMBIO DE PASSWD

# NORMALMENTE SE DEFINE UN MAXIMO O UN MINIMO DE DIAS PERO NO LOS DOS PORQUE GENERA CONTRADICCION

# BLOQUEAR Y DESBLOQUEAR PASSWD
usermod -L 
usermod -U

# ESTABLECER PARAMETROS DE ENVEJECIMIENTO DE PASSWORD PARA EL USUARIO ocantoran DE LA SIGUIENTE MANERA: mindays 7 maxdays 28 werndays 5
chage -m 7 -M 28 -W 5 ocantoran
chage -l ocantoran


# ESTABLECER LA FECHA DE EXPIRACION DE LA CUENTA DE USUARIO PARA 30 DE JUNIO 2022
chage -E 2022-06-30 ocantoran

# ESTABLECER PARAMETROS CON EL COMANDO PASSWD PARA EL USUARIO ocantoran DE LA SIGUIENTE MANERA mindays 10 maxdays 90 warndays 14
passwd -n 10 -x 90 -w 14 ocantoran

# CONFIRMAR PARAMETROS 
passwd -S ocantoran

# ESTABLECER DIAS DE INACTIVIDAD (DIAS DESPUES DE CADUCADA EL PASSWD LA CUENTA QUEDA INACTIVA)
passwd -i 5 ocantoran

# EL USUARIO DEBE CAMBIAR EL PASSWD EN EL SIGUIENTE INICIO DE SESION 
passwd -e ocantoran 

# PROCEDA A BLOQUEAR LA CUENTA DEL USUARIO ocantoran
passwd -l ocantoran
usermod -L ocantoran

# DESBLOQUEAR
passwd -u ocantoran
usermod -U ocantoran


# VER CUAL ES EL USUARIO INICIAL
logname
